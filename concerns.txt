// mktime usage 

#include <ctime>
#include <iomanip>
#include <iostream>
#include <sstream>
 
int main()
{
    setenv("TZ", "/usr/share/zoneinfo/America/Los_Angeles", 1); // POSIX-specific
 
    std::tm tm{}; // Zero initialise
    tm.tm_year = 2020 - 1900; // 2020
    tm.tm_mon = 2 - 1; // February
    tm.tm_mday = 15; // 15th
    tm.tm_hour = 10;
    tm.tm_min = 15;
    tm.tm_isdst = 0; // Not daylight saving
    std::time_t t = std::mktime(&tm); 
    std::tm local = *std::localtime(&t);
 
    std::cout << "local: " << std::put_time(&local, "%c %Z") << '\n';
}

// Another method to Calculate difference between two times

#include <iostream>
#include <chrono>
#include <sstream>
#include <iomanip>
#include <string>

int main() {
    // Define start and end times as strings
    std::string start_time = "Wed Apr  3 14:17:22 2024";
    std::string end_time = "Wed Apr  3 23:47:22 2024";
    
    // Define time format
    std::locale loc("C");
    std::istringstream start_ss(start_time), end_ss(end_time);
    start_ss.imbue(loc);
    end_ss.imbue(loc);

    // Parse start and end times
    std::tm start_tm = {}, end_tm = {};
    start_ss >> std::get_time(&start_tm, "%a %b %e %T %Y");
    end_ss >> std::get_time(&end_tm, "%a %b %e %T %Y");

    // Convert parsed times to std::chrono::time_point
    auto start_point = std::chrono::system_clock::from_time_t(std::mktime(&start_tm));
    auto end_point = std::chrono::system_clock::from_time_t(std::mktime(&end_tm));

    // Calculate the difference
    auto difference = end_point - start_point;

    // Convert the difference to seconds
    auto seconds = std::chrono::duration_cast<std::chrono::seconds>(difference).count();

    std::cout << "\nDifference in time is: " << static_cast<double>(seconds)/3600 << " hours.\n" << std::endl;

    return 0;
}
